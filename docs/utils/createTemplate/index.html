createTemplate
утилита для создание HTML элементов
import { createTemplate } from 'jcore-ui';
import createTemplate from 'jcore-ui/bin/lib/utils/createTemplate';
Использование
Функция должна принимать только 1 родительский элемент

const data = [
    { id: 1, title: 'first' },
    { id: 2, title: 'second' }
];

const $element = createTemplate(`
    

        ${data.map(item => {
            return `
                <div class="custom-list-item" data-id="${item.id}">${item.title}</div>
            `;
        }).join('')}
    

`);
